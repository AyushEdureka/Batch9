    <div id="mydiv">
        <div class="content">Scroll inside me</div>

    <script>
        //select element
        const button = document.getElementById("mybutton")
        const input = document.getElementById("myInput")
        const div = document.getElementById("mydiv")


        //1 click events

        button.addEventListener('dblclick',function(){
            alert('Product added to cart')
        })

        //2nd Way
        
         function handleKeup(event){
            console.log('key released :'+  event.key )
         }
         input.addEventListener('keyup',handleKeup)


         //change: help us to detect the change in input
         function handleChange(e){
            console.log('Final Output ' + e.target.value)
         }

         input.addEventListener('change',handleChange)


         function handleScroll(){
            console.log('Scrolled')
         }
         div.addEventListener('scroll',handleScroll)

         function handleMouseOver(){
               const div = document.querySelector('#mydiv')
               div.style.backgroundColor = 'red'
         }
         div.addEventListener('mouseover',handleMouseOver)


         function handleMouseOut(){
               const div = document.querySelector('#mydiv')
               div.style.backgroundColor = 'green'
            
         }
         div.addEventListener('mouseout',handleMouseOut)


    </script>
